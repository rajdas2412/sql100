select * from entries;


-- Rank the count of visits sorted in desc count
-- Pick the top row 
select name, floor most_visited_floor  
from(
	select name, floor,
	rank() over(partition by name order by count(floor) desc) as visit_rank
	from entries
	group by name, floor
) where visit_rank=1;

-- Resources used
select name, count(floor) total_visits, string_agg(resources, ',')
from entries
group by name;


-- CTE for most visits and resources used
with most_visits as (
select name, floor most_visited_floor  
from(
	select name, floor,
	rank() over(partition by name order by count(floor) desc) as visit_rank
	from entries
	group by name, floor
) where visit_rank=1 
), resource_used as (
select name, count(floor) total_visits, string_agg(resources, ',') resources_used
from entries
group by name
)
select m.name, m.most_visited_floor,
r.total_visits, r.resources_used
from most_visits m join resource_used r
on m.name = r.name
